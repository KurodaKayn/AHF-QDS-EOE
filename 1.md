# 结构
```
src/
├── app/                # Next.js 路由与页面（主业务入口）
│   ├── api/            # API 路由（服务端接口，数据库交互）
│   ├── login/          # 登录页面
│   ├── register/       # 注册页面
│   ├── quiz/           # 题库、练习、复习、设置等页面
│   └── ...             # 其他页面与全局布局
├── components/         # 可复用UI组件与业务组件
│   ├── ui/             # 基础UI组件（Button、Dialog等）
│   ├── quiz/           # 题库相关组件
│   └── ...             # 其他弹窗、表单等
├── constants/          # 常量与配置
├── generated/          # 自动生成代码（如prisma类型）
│   └── prisma/
├── hooks/              # 自定义React hooks
├── lib/                # 工具库与通用逻辑
│   ├── db/             # 数据库操作封装
│   ├── api/            # API封装
│   └── ...             # AI、用户设置等
├── store/              # 状态管理（如quizStore、themeStore）
├── types/              # TypeScript类型定义
├── utils/              # 工具函数
└── middleware.ts       # 中间件
```
